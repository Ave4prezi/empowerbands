# Activation routes for managing band activations
# This module handles the /activate endpoint and full profile activation

from flask import Blueprint, request, jsonify
from datetime import datetime

activation_bp = Blueprint('activation', __name__)

@activation_bp.route('/activate', methods=['POST'])
def activate_profile():
    """
    Activate the full user profile
    Required JSON payload:
    {
        "user_id": "string",
        "bands": ["band_id_1", "band_id_2", ...]
    }
    """
    data = request.get_json()
    
    if not data or 'user_id' not in data:
        return jsonify({'error': 'Missing user_id'}), 400
    
    user_id = data.get('user_id')
    bands = data.get('bands', [])
    
    # Activate profile logic
    activation_record = {
        'user_id': user_id,
        'bands': bands,
        'activated_at': datetime.now().isoformat(),
        'status': 'active'
    }
    
    return jsonify({
        'message': 'Profile activated successfully',
        'activation': activation_record
    }), 200
